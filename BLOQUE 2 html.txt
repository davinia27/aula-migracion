    function drawWheel() {
        const c = document.getElementById('wheel'); const ctx = c.getContext("2d");
        const ratio = window.devicePixelRatio || 2;
        c.width = 1000 * ratio; c.height = 1000 * ratio; ctx.scale(ratio, ratio);
        const center = 500; const slice = (2 * Math.PI) / countries.length;
        countries.forEach((p, i) => {
            ctx.beginPath(); ctx.moveTo(center, center); ctx.arc(center, center, 480, i*slice, (i+1)*slice);
            ctx.fillStyle = p.c; ctx.fill(); ctx.lineWidth = 12; ctx.strokeStyle = "#000"; ctx.stroke();
            ctx.save(); ctx.translate(center, center); ctx.rotate(i*slice + slice/2);
            ctx.fillStyle = "white"; ctx.font = "900 65px Plus Jakarta Sans"; ctx.textAlign = "right";
            ctx.fillText(p.n, 460, 20); ctx.restore();
        });
    }

    function spinWheel() {
        if (spinCount >= 5) return;
        const spinBtn = document.getElementById('spinBtn');
        spinBtn.disabled = true;
        let avail = countries.map((_, i) => i).filter(idx => !assignedC.includes(idx));
        let targetIdx = avail[Math.floor(Math.random() * avail.length)];
        const seg = 360 / countries.length; 
        rot += 360 * 10 + (270 - (targetIdx * seg + seg/2)) - (rot % 360);
        document.getElementById('wheel').style.transform = `rotate(${rot}deg)`;
        setTimeout(() => { 
            assignedC[spinCount] = targetIdx; 
            updateWheelMini(); 
            confetti(); 
            spinCount++;
            if (spinCount < 5) {
                document.getElementById('spinStatus').innerText = `Sorteando: GRUPO ${spinCount + 1}`;
                spinBtn.disabled = false;
            } else {
                document.getElementById('spinStatus').innerText = "Sorteo Finalizado";
            }
        }, 6000);
    }

    async function generarIdentidadIA() {
        try {
            const prompt = `Genera un nombre de despacho legal y una filosofía breve para un grupo de expertos en extranjería. JSON: {"n":"", "f":""}`;
            const url = `https://generativelanguage.googleapis.com/v1beta/models/${modelId}:generateContent?key=${apiKey}`;
            const res = await fetch(url, { method: 'POST', headers: {'Content-Type': 'application/json'}, body: JSON.stringify({ contents: [{ parts: [{ text: prompt }] }] }) });
            const data = await res.json();
            const json = JSON.parse(data.candidates[0].content.parts[0].text.match(/\{.*\}/s)[0]);
            document.getElementById('gName').innerText = json.n;
            document.getElementById('gMotive').innerText = json.f;
        } catch(e) {}
    }

        const c = sessionConcepts[currentS][cI];
        const opts = el.parentElement.querySelectorAll('.concept-option');

        if(solvedCount >= 4) {
            if(currentS === 1 && !unlocked.includes(2)) unlocked.push(2);
            if(currentS === 2 && !unlocked.includes(3)) unlocked.push(3);
            updateLocks(); confetti();
        }
    }
